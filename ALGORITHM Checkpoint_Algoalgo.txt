VAR
    c : CHAR;
    n_char, n_wor, n_vow : INTEGER;
BEGIN
    n_char:=0;
    n_vow:=0;
    n_wor:=0;
    write("Entrer le caract√®re :" );
    read(c);
    WHILE (c<>".") DO
        n_char+=1
        IF (c in ["a", "e", "i", "o", "u", "y"]) THEN

        END_IF
        IF (c=" ") THEN

    END_WHILE

END
